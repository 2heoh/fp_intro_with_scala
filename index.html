Functions

// regular function
def adder(a: Int, b: Int): Int =  {
  a + b
}
// or
def adder(a: Int, b: Int) = a + b

adder(2, 2) // Int = 4

val lambda = (a: Int, b: Int) => a + b
lambda(2, 2) // Int = 4

Pure functions

//has side-effects
def shout(word: String): Unit = {
  println(word + "!")
}
shout("hey") // "hey!"

//no side-effects
def pureShout(word: String): String = word + "!"
print(pureShout("hey")) // "hey!"

Recursion

def factorial(n: Int): Int = 
  if(n > 0) factorial(n-1) * n else 1

factorial(10) // 3628800

factorial(100000) //java.lang.StackOverflowError

Tail recursion

def betterFactorial(x: Int) = {
    // auxiliary function				    
    def loop(n: Int, acc: Int): Int = 
      if(n <= 0) 1 else loop(n-1, acc*n) 

    loop(x-1, x)
}

betterFactorial(100000) //

Pattern matching

def bestFactorial(n: Int, acc: Int = 0): Int = n match{
    case 0 => 1
    case n => bestFactorial(n-1, acc*n)
}

bestFactorial(100000) //
