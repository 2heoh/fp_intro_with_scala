Functions

// regular function
def adder(a: Int, b: Int): Int =  {
  a + b
}
// or
def adder(a: Int, b: Int) = a + b

adder(2, 2) // Int = 4

// anonymous function
val f = (a: Int, b: Int) => a + b
f(2, 2) // Int = 4

Pure functions

//has side-effects
def shout(word: String): Unit = {
  println(word + "!")
}
shout("hey") // "hey!"

//no side-effects
def pureShout(word: String): String = word + "!"
print(pureShout("hey")) // "hey!"

Type system

// Static strong type system
def tl(l: String) = l.toList() match{
	case Nil 	=> throw new Exception("empty list")
	case a::as 	=> as
}
